* Advanced Settings
** Functions
#+begin_src emacs-lisp
  (defun shrink-left ()
    (interactive)
    (enlarge-window-horizontally 10))

  (defun shrink-right ()
    (interactive)
    (shrink-window-horizontally 10))

  (defun update-backend ()
    (interactive)
    (org-babel-tangle)
    (compile "make -k"))

  (defun org-agenda-and-todo ()
    (interactive)
    (org-agenda nil "n"))

  (defun add-task-to-org ()
    (interactive)
    (org-capture nil "t"))
#+end_src
